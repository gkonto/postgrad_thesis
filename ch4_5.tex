int main(int argc, char **argv) {
    Opts o;
    parseArgs(argc, argv, o);

    Node *head = init_nodes(o.num_nodes_, FIBVAL);
    Node *p = head;

    double start = omp_get_wtime();
    while(p != nullptr) {
        fib(p->data_);
        p = p->next_;
    }
    double end = omp_get_wtime();

    delete head;

    std::cout << "Time Elapsed: " << end - start << std::endl;
    return 0;
}
